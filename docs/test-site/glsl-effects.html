1    GLSL Effects

1.1    Introduction

The :shader directive embeds a GLSL effect from src-lang/glsl/effects/ and renders it as a syntax-highlighted source block with a live WebGL preview.

1.2    Solid Color

A flat color effect.

void main(){
  gl_FragColor = vec4(0.2,0.6,0.9,1.0);
}

1.3    Gradient

A horizontal gradient based on fragment x coordinate.

uniform vec2 u_resolution;
void main(){
  float t = gl_FragCoord.x / u_resolution.x;
  gl_FragColor = vec4(t,1.0 - t,0.2,1.0);
}

1.4    Checkerboard

An 8×8 checker pattern.

uniform vec2 u_resolution;
void main(){
  float cx = floor((8.0 * gl_FragCoord.x) / u_resolution.x);
  float cy = floor((8.0 * gl_FragCoord.y) / u_resolution.y);
  float c = fract(0.5 * (cx + cy));
  float v = step(0.5,c);
  gl_FragColor = vec4(v,v,v,1.0);
}

1.5    Noise

Pseudo-random value noise from screen coordinates.

void main(){
  float v = fract(
    sin((gl_FragCoord.x * 12.9898) + (gl_FragCoord.y * 78.233)) * 43758.5453
  );
  gl_FragColor = vec4(v,v,v,1.0);
}

1.6    Plasma

Animated sine/cosine plasma.

uniform vec2 u_resolution;
uniform float u_time;
void main(){
  float x = gl_FragCoord.x / u_resolution.x;
  float y = gl_FragCoord.y / u_resolution.y;
  float v = sin(x * 10.0) + cos(y * 10.0) + sin(x + y + u_time);
  gl_FragColor = vec4(
    0.5 + (0.5 * sin(v)),
    0.5 + (0.5 * cos(v)),
    0.5 + (0.5 * sin(v + 1.0)),
    1.0
  );
}

1.7    Ripple

A distance ripple from the center of the canvas.

uniform vec2 u_resolution;
uniform float u_time;
void main(){
  float cx = gl_FragCoord.x - (0.5 * u_resolution.x);
  float cy = gl_FragCoord.y - (0.5 * u_resolution.y);
  float d = sqrt((cx * cx) + (cy * cy));
  float v = sin((d * 0.2) + u_time);
  gl_FragColor = vec4(0.5 + (0.5 * v),0.5 + (0.5 * v),1.0,1.0);
}
